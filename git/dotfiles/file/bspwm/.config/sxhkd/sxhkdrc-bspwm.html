#
## my hotkeys
#

# Lock screen
super + shift + x
  lock.sh

XF86AudioRaiseVolume
  amixer sset Master 5%+ unmute

XF86AudioLowerVolume
  amixer sset Master 5%- unmute

XF86AudioMute
  amixer sset Master toggle

XF86MonBrightnessUp
  xbacklight -inc 10%

XF86MonBrightnessDown
  xbacklight -dec 10%

XF86AudioPlay
  mpc toggle

XF86AudioStop
  mpc stop

XF86AudioNext
  mpc next

XF86AudioPrev
  mpc prev

#
## bspwm hotkeys
#

# Quit bspwm and kill the panel
super + alt + Escape
#~/.scripts/exitbspwm
  pkill panel; bspc quit 1
super + alt + shift + Escape
  pkill panel; bspc wm -d > "$BSPWM_STATE"; bspc quit
# Close the focused window
super + w
	bspc node -c

# Toggle monocle layout
super + n
	bspc desktop -l next

# Change selected window's mode into tiled, pseudo-tiled, floating or fullscreen
super + {t,p,s,f}
	bspc node -t {tiled,pseudo_tiled,floating,fullscreen}

# Focus the last window/desktop
super + {grave,Tab}
	bspc {node,desktop} -f last

# Swap the focused window with the last window
super + apostrophe
	bspc node -s last

# Focus the next older/newer window
super + {o,i}
	bspc control --record-history off; \
	bspc node {older,newer} -f; \
	bspc control --record-history on


super + y
	bspc node -n last.!automatic

super + m
	bspc node -s biggest


super + {_,shift + }{h,j,k,l}
	bspc node -{f,s} {west,south,north,east}

super + {_,shift + }c
	bspc node -f {next,prev}

super + {comma,period}
	bspc desktop -C {backward,forward}

super + bracket{left,right}
	bspc desktop -f {prev,next}

super + ctrl + {h,j,k,l}
	bspc node -p {west,south,north,east}

super + ctrl + {_,shift + }space
	bspc {node -p cancel,desktop -c}

# expand the tiled space in the given direction
super + alt + {h,j,k,l}
	bspc node {@west -r -10,@south -r +10,@north -r -10,@east -r +10}

# contract the tiled space in the given direction
super + alt + shift + {h,j,k,l}
	bspc node {@east -r -10,@north -r +10,@south -r -10,@west -r +10}

super + ctrl + {1-9}
	bspc node -r 0.{1-9}

super + {Left,Down,Up,Right}
	xdo move {-x -20,-y +20,-y -20,-x +20}

super + {_,shift + }{1-9,0}
	bspc {desktop -f,node -d} '^{1-9,10}'

super + shift + s
  bspc node -g sticky

~button1
	bspc pointer -g focus

super + button{1-3}
	; bspc pointer -g {move,resize_side,resize_corner}

super + @button{1-3}
	bspc pointer -u
