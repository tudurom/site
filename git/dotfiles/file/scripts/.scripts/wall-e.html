#! /bin/sh -x
#
# Set wallpaper
# (c) tudurom 2016 - wtfpl
#

DEFAULT_WALL="$HOME/wallpapers/venice.jpg"
APRIL_WALL="$HOME/wallpapers/april_wall.png"

if [ "$(date +"%d/%m")" = "01/04" ]; then
  feh -fill "$APRIL_WALL"
elif [ $# -ge 1 ]; then
  option="fill"
  test -n "$2" && option="$2"
  feh --bg-${option} $1
else
  feh --bg-fill $DEFAULT_WALL
fi
