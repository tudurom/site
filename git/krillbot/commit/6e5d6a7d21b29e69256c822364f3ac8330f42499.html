+func commandKill(u tgbotapi.Update, b *tgbotapi.BotAPI) {
+	killmsgs := getKillMsgs()
+	rand.Seed(int64(u.Message.MessageID))
+	str := fmt.Sprintf(killmsgs[rand.Intn(len(killmsgs))], u.Message.CommandArguments())
+
+	// Send a hypnotoad if no params given
+	if u.Message.CommandArguments() == "" {
+		str = fmt.Sprintf("all glory to the hypnotoad")
+
+		var hypnotoad tgbotapi.DocumentConfig
+		hypnotoad = tgbotapi.NewDocumentUpload(u.Message.Chat.ID, Config.HypnoPath)
+
+		b.Send(hypnotoad)
+	} else {
+		msg := tgbotapi.NewMessage(u.Message.Chat.ID, str)
+
+		b.Send(msg)
+	}
+}
+
+func commandSuicide(u tgbotapi.Update, b *tgbotapi.BotAPI) {
+	killmsgs := getKillMsgs()
+	rand.Seed(int64(u.Message.MessageID))
+	str := fmt.Sprintf(killmsgs[rand.Intn(len(killmsgs))], u.Message.From.FirstName)
+
+	// Add a reason if available
+	if u.Message.CommandArguments() != "" {
+		str += ". Reason: " + u.Message.CommandArguments() + "."
+	}
+
+	msg := tgbotapi.NewMessage(u.Message.Chat.ID, str)
+	b.Send(msg)
+}

+package main
+
+import (
+	"encoding/json"
+	"flag"
+	"io/ioutil"
+	"log"
+	"os"
+
+	"gopkg.in/telegram-bot-api.v4"
+)
+
+type ConfigStructure struct {
+	Token     string
+	Debug     bool
+	HypnoPath string `json:"hypnopath"`
+}
+
+type command func(tgbotapi.Update, *tgbotapi.BotAPI)
+
+var Logger *log.Logger
+var Config ConfigStructure
+var Bot tgbotapi.BotAPI
+var Commands map[string]command
+
+func initLogger() *log.Logger {
+	return log.New(os.Stdout, "", log.Lshortfile)
+}
+
+func initCommands() {
+	Commands["kill"] = commandKill
+	Commands["suicide"] = commandSuicide
+	Commands["bleach"] = commandBleach
+	Commands["lenny"] = commandLenny
+	Commands["flip"] = commandFlip
+}
+
+func main() {
+	var configFile = flag.String("config", "config.json", "path to the config file")
+
+	Logger = initLogger()
+
+	// Read config file
+	flag.Parse()
+	rawData, err := ioutil.ReadFile(*configFile)
+	if err != nil {
+		Logger.Fatalf("Error while reading config file: %s", err)
+	}
+
+	err = json.Unmarshal(rawData, &Config)
+	if err != nil {
+		Logger.Fatalf("Error while decoding config file: %s", err)
+	}
+	Logger.Printf("Config: %s\n", Config)
+
+	// Connect to telegram
+	Bot, err := tgbotapi.NewBotAPI(Config.Token)
+	if err != nil {
+		Logger.Fatalf("Error while creating Bot instance: %s", err)
+	}
+	Bot.Debug = Config.Debug
+
+	u := tgbotapi.NewUpdate(0)
+	u.Timeout = 60
+
+	updates, err := Bot.GetUpdatesChan(u)
+
+	Commands = make(map[string]command)
+	initCommands()
+
+	for update := range updates {
+		if update.Message == nil {
+			continue
+		}
+
+		// Reply to commands
+		if update.Message.Command() != "" && Commands[update.Message.Command()] != nil {
+			Commands[update.Message.Command()](update, Bot)
+		}
+	}
+}
