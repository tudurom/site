#include <stdio.h>
#include <stdlib.h>
#include <err.h>
#include <locale.h>

#define SMALLEST_VERTICAL_CHAR L'▁'
#define SMALLEST_HORIZONTAL_CHAR L'▏'
#define VERTICAL +1
#define HORIZONTAL -1
#define SIZE_UNITS 8
#define DEFAULT_SIZE 10 * SIZE_UNITS
#define DEFAULT_ORIENTATION VERTICAL
#define DEFAULT_SPACE_CHAR ' '

static int orientation, size, real_size;
static char space_char;

static void print_spaces();
static void print_middle_block();
static void print_blocks();

int
main(int argc, char **argv)
{
    int value;

    size = getenv("SIZE") ? atoi(getenv("SIZE")) : DEFAULT_SIZE;
    space_char = getenv("SPACE") ? getenv("SPACE")[0] : DEFAULT_SPACE_CHAR;

    if (argc > 1)
        value = atoi(argv[1]);
    else
        scanf("%d", &value);

    if (getenv("ORIENT")) {
        if (getenv("ORIENT")[0] == '|')
            orientation = VERTICAL;
        else if (getenv("ORIENT")[0] == '-')
            orientation = HORIZONTAL;
        else
            errx(1, "incorrect orientation value");
    } else {
        orientation = DEFAULT_ORIENTATION;
    }

    real_size = (value * size) / 100;

    setlocale(LC_ALL, "");
    if (orientation == VERTICAL) {
        print_spaces();
        print_middle_block();
        print_blocks();
    } else {
        print_blocks();
        print_middle_block();
        print_spaces();
    }

    return 0;
}

/*
 * Fill empty blocks.
 */

static void
print_spaces()
{
    for (int i = 0; i < (size - real_size) / SIZE_UNITS; i++) {
        if (orientation == VERTICAL)
            fputc('\n', stdout);
        else
            fputc(space_char, stdout);
    }
}

/*
 * Print the variable-width block.
 */

static void
print_middle_block()
{
    if (real_size % SIZE_UNITS > 0) {
        if (orientation == VERTICAL)
            printf("%lc\n", SMALLEST_VERTICAL_CHAR + real_size % SIZE_UNITS - 1);
        else
            printf("%lc", SMALLEST_HORIZONTAL_CHAR - real_size % SIZE_UNITS + 1);
    }
}

/*
 * Print full-width blocks.
 */

static void print_blocks()
{
    for (int i = 0; i < real_size / SIZE_UNITS; i++) {
        printf("%lc", SMALLEST_VERTICAL_CHAR + SIZE_UNITS - 1);
        if (orientation == VERTICAL)
            fputc('\n', stdout);
    }
}
