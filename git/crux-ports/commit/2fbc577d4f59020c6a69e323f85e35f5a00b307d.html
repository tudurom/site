+# Description: A tiny launcher menu packing a big bang
+# URL: https://github.com/TrilbyWhite/interrobang
+# Packager: arcetera, echo iue@trnspljc.890 | tr ietursnpl890jc acprlsurecomed
+# Depends: xorg-libx11 git
+
+name=interrobang
+version=git
+release=1
+source=()
+
+build() {
+  cd $PKGMK_SOURCE_DIR
+
+  if cd $name; then
+    git fetch -q; git reset --hard origin/master
+  else
+    git clone git://github.com/TrilbyWhite/interrobang.git
+    cd $name
+  fi
+
+  make
+  make PREFIX=/usr DESTDIR=$PKG install
+}
