+# Description: Simple terminal with Xresources support.
+# URL:         https://github.com/dcat/st-xresources
+# Packager:    arcetera, arcetera@cock.li
+# Depends on:  git xorg-libx11
+
+name=st-xresources
+version=git
+release=1
+source=(config.h)
+
+build() {
+	cd $PKGMK_SOURCE_DIR
+
+	if cd $name; then
+		git fetch -q; git reset --hard origin/master
+	else
+		git clone https://github.com/dcat/st-xresources.git -b master $name
+		cd $name
+	fi
+	
+	mv $SRC/config.h .
+	
+	make
+	make DESTDIR=$PKG install
+	tic -s st.info
+}
