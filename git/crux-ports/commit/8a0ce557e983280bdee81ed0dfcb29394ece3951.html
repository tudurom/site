+# Description: Perl Compatible Regular Expressions
+# URL: http://www.pcre.org
+# Maintainer: Jose V Beneyto, sepen at crux dot nu
+# Packager: Jose V Beneyto, sepen at crux dot nu
+# Depends on:
+
+name=libpcre2
+version=10.21
+release=2
+source=(ftp://ftp.csx.cam.ac.uk/pub/software/programming/pcre/pcre2-$version.tar.bz2)
+
+build() {
+    cd pcre2-$version
+
+    ./configure --prefix=/usr \
+                --enable-jit  \
+                --enable-pcre2-32
+
+    make
+    make DESTDIR=$PKG install
+
+    rm -r $PKG/usr/share/doc
+}

+# Description: CRUX port utilities without prt-get usage written in fish.
+# URL:         https://github.com/onodera-punpun/prt-stuff
+# Maintainer:  6c37 Team, https://github.com/6c37/crux-ports/issues
+# Depends on:  git getopts
+
+name=prtstuff
+version=git
+release=1
+source=()
+
+build () {
+	cd $PKGMK_SOURCE_DIR
+
+	if cd $name; then
+		git fetch -q; git reset --hard origin/master
+	else
+		git clone git://github.com/onodera-punpun/prtstuff $name
+		cd $name
+	fi
+
+	make DESTDIR=$PKG install
+}
