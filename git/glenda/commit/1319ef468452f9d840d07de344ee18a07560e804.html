+#include <stdio.h>
+#include <stdlib.h>
+#include <time.h>
+
+/* DEBUG */
+#ifndef D
+#define D 0
+#endif
+/* NO. of colors */
+#define N_COL 5
+/* Maximum size of the board */
+#define MAX_W 50
+#define MAX_H 50
+/* Player values. S is up (sus) and J is down (jos) */
+#define S -1
+#define J 1
+
+typedef char color;
+
+color board[MAX_H][MAX_W];
+color colors[] = {'@', '#', '+', '.', '*'};
+color player;
+
+/* Flood fill old color with new one */
+void fill(int lin, int col, int n, int m, color old, color new) {
+    board[lin][col] = new;
+    if (lin > 0     && board[lin - 1][col] == old)
+        fill(lin - 1, col, n, m, old, new);
+    if (lin < n - 1 && board[lin + 1][col] == old)
+        fill(lin + 1, col, n, m, old, new);
+    if (col > 0     && board[lin][col - 1] == old)
+        fill(lin, col - 1, n, m, old, new);
+    if (col < m - 1 && board[lin][col + 1] == old)
+        fill(lin, col + 1, n, m, old, new);
+}
+
+int main() {
+    int lin, col;
+    int n, m;
+    color c;
+    /* colors of the current player and opponent */
+    color me, op;
+    int i;
+    int pos;
+
+    /* Reopen stdin and stdout to files for easier debugging */
+    if (D) {
+        freopen("glenda.in", "r", stdin);
+        freopen("glenda.out", "w", stdout);
+    }
+
+    player = fgetc(stdin);
+    fgetc(stdin);
+    if (player == 'J')
+        player = J;
+    else
+        player = S;
+
+    c = fgetc(stdin);
+    lin = col = 0;
+    while (c != EOF) {
+        if (c == '\n') {
+            m = col;
+            lin++;
+            col = 0;
+        } else {
+            board[lin][col] = c;
+            col++;
+        }
+        c = fgetc(stdin);
+    }
+    n = lin;
+    if (player == J) {
+        me = board[n - 1][0];
+        op = board[0][m - 1];
+    } else {
+        me = board[0][m - 1];
+        op = board[n - 1][0];
+    }
+
+    /* Remove colors used by the player and the opponent */
+    for (i = 0; i < N_COL; i++) {
+        if (colors[i] == me || colors[i] == op)
+            colors[i] = -1;
+    }
+
+    /* Get color to play */
+    srand(time(NULL));
+    pos = rand() % N_COL;
+    while (colors[pos] == -1)
+        pos = (pos + 1) % N_COL;
+
+    /* do the move */
+    if (player == S) {
+        lin = 0;
+        col = m - 1;
+    } else {
+        lin = n - 1;
+        col = 0;
+    }
+    fill(lin, col, n, m, board[lin][col], colors[pos]);
+
+    /* print the board */
+    if (player == J)
+        fputc('S', stdout);
+    else
+        fputc('J', stdout);
+    fputc('\n', stdout);
+    for (lin = 0; lin < n; lin++) {
+        for (col = 0; col < m; col++)
+            fputc(board[lin][col], stdout);
+        fputc('\n', stdout);
+    }
+
+    /* Closes files just in case D flag is 1 */
+    fclose(stdin);
+    fclose(stdout);
+    return 0;
+}
