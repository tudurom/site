#ifdef D
#include <assert.h>
#endif
#include <limits.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

/* Constants {{{ */
#ifdef D
#define debug(...) fprintf(stderr, __VA_ARGS__)
#define crash(...) assert(__VA_ARGS__)
#else
#define debug(...)
#define crash(...)
#endif

#define MAX_LIN 50
#define MAX_COL 50

#define COLORS 5

#define DEPTH 8

#define S -1
#define J +1

#define MATERIAL_WEIGHT 10
#define AREA_WEIGHT 2
#define DIST_WEIGHT 1

#define DIRECTIONS 4

#define BORDER SCHAR_MIN
#define INFINITY (SCHAR_MAX - 1)
#define INT_INFINITY (SHRT_MAX - 1)

#define max(a, b) (a > b ? a : b)
#define min(a, b) (a < b ? a : b)
/* }}} */

/* Globals {{{ */
char board[MAX_LIN + 2][MAX_COL + 2];
char alt_board[MAX_LIN + 2][MAX_COL + 2];
int board_lines, board_cols;

unsigned char colors[COLORS] = {'@', '#', '+', '.', '*'};

short fill_counter = 0;

unsigned char positional_color;
char positional_corner;

char limit_v, limit_h;

char dist_min;

char current_player;
/* }}} */

/* Functions {{{ */
/* Return a player's corner position {{{ */
/* S is up ("sus"), J is down ("jos") */
inline void get_corner(char current_player, unsigned char *lin, unsigned char *col) {
    if (current_player == S) {
        *lin = 0;
        *col = board_cols - 1;
    } else {
        *lin = board_lines - 1;
        *col = 0;
    }
}
/* }}} */

/* Initialize fill variables {{{ */
inline void prepare_fill() {
    fill_counter = 0;
}

inline void prepare_positional(unsigned char color, char corner) {
    fill_counter = 0;
    positional_color = color;
    positional_corner = corner;
    limit_v = 0;
    limit_h = 0;
    dist_min = INFINITY;
}
/* }}} */

/* Replace and fill at specified coordinates {{{ */
/* Stores the amount of filled squares in fill_counter */
void fill(char (*myboard)[MAX_COL + 2], unsigned char lin, unsigned char col, unsigned char color_old, unsigned char color_new) {
    myboard[lin + 1][col + 1] = color_new;
    fill_counter++;
    if (myboard[1 + lin - 1][1 + col] == color_old)
        fill(myboard, lin - 1, col, color_old, color_new);
    if (myboard[1 + lin + 1][1 + col] == color_old)
        fill(myboard, lin + 1, col, color_old, color_new);
    if (myboard[1 + lin][1 + col - 1] == color_old)
        fill(myboard, lin, col - 1, color_old, color_new);
    if (myboard[1 + lin][1 + col + 1] == color_old)
        fill(myboard, lin, col + 1, color_old, color_new);
}
/* }}} */

/* Return the number of squares a player has {{{ */
inline short get_material_score(char (*myboard)[MAX_COL + 2], unsigned char lin, unsigned char col) {
    char score_board[MAX_LIN + 2][MAX_COL + 2];

    /* Copy to temp board */
    memcpy(score_board, myboard, sizeof(score_board));

    /* Fill with infinity */
    prepare_fill();
    fill(score_board, lin, col, score_board[lin + 1][col + 1], INFINITY);

    return fill_counter;
}
/* }}} */

/* Get positional score {{{ */
void get_positional_score(char (*myboard)[MAX_COL + 2], unsigned char lin, unsigned char col) {
    unsigned char op_x, op_y;
    char v, h;
    /* don't go back */
    fill_counter++;
    myboard[lin + 1][col + 1] = INFINITY;

    /* v is the vertical limit, h is the horizontal limit */
    if (positional_corner == J) {
        v = board_lines - 1 - lin;
        h = col;
        op_x = board_cols - 1;
        op_y = 0;
    } else {
        v = lin;
        h = board_cols - 1 - col;
        op_x = 0;
        op_y = board_lines - 1;
    }

    /* set maximum limits */
    if (v > limit_v)
        limit_v = v;
    if (h > limit_h)
        limit_h = h;
    /* manhattan distance from point to opposite corner */
    if (abs(lin - op_y) + abs(col - op_x) < dist_min)
        dist_min = abs(lin - op_y) + abs(col - op_x);

    if (myboard[1 + lin - 1][1 + col] == positional_color)
        get_positional_score(myboard, lin - 1, col);
    if (myboard[1 + lin + 1][1 + col] == positional_color)
        get_positional_score(myboard, lin + 1, col);
    if (myboard[1 + lin][1 + col - 1] == positional_color)
        get_positional_score(myboard, lin, col - 1);
    if (myboard[1 + lin][1 + col + 1] == positional_color)
        get_positional_score(myboard, lin, col + 1);
}
/* }}} */

/* Make a border around the board {{{ */
inline void init_board(char (*myboard)[MAX_COL + 2]) {
    int i, j;
    /* magic value */
    int init = BORDER;
    debug("initializing board with %d\n", -init);
    for (i = 0; i <= 1; i++)
        for (j = 0; j < board_cols + 2; j++)
            myboard[i * (board_lines + 2 - 1)][j] = -init;

    for (j = 0; j <= 1; j++)
        for (i = 0; i < board_lines + 2; i++)
            myboard[i][j * (board_cols + 2 - 1)] = -init;
}
/* }}} */

/* Utils {{{ */
inline unsigned char char_to_color(unsigned char char_color) {
    switch (char_color) {
        case '@': return 0;
        case '#': return 1;
        case '+': return 2;
        case '.': return 3;
        case '*': return 4;
    }

    /* if returned nothing yet, crash */
    debug("attempt to convert char %c to color\n", char_color);
    crash(1 == 0);
    return 0;
}

inline unsigned char color_to_char(unsigned char color_char) {
    switch (color_char) {
        case 0: return '@';
        case 1: return '#';
        case 2: return '+';
        case 3: return '.';
        case 4: return '*';
    }

    /* same as above */
    debug("attempt to convert color %d to char\n", color_char);
    crash(1 == 0);
    return '@';
}
/* }}} */

/* Minimax {{{ */
short minimax(char (*myboard)[MAX_COL + 2], unsigned char depth, char player, short alpha, short beta) {
    char copy_board[MAX_LIN + 2][MAX_COL + 2];
    unsigned char me_lin, me_col, op_lin, op_col;
    unsigned char me, op;
    unsigned char color;
    short score, val;

    memcpy(copy_board, myboard, sizeof(copy_board));

    get_corner(player, &me_lin, &me_col);
    get_corner(-player, &op_lin, &op_col);

    /* calculate score for the above table in the tree */
    if (depth == 0) {
        prepare_positional(myboard[me_lin + 1][me_col + 1], player);
        get_positional_score(copy_board, me_lin, me_col);
        score = MATERIAL_WEIGHT * (fill_counter - get_material_score(copy_board, op_lin, op_col));
        score += DIST_WEIGHT * dist_min;
        score += AREA_WEIGHT * (limit_v + 1) * (limit_h + 1);
        score /= MATERIAL_WEIGHT + DIST_WEIGHT + AREA_WEIGHT;

        return score;
    }

    me = myboard[me_lin + 1][me_col + 1];
    op = myboard[op_lin + 1][op_col + 1];
    score = -INT_INFINITY;
    color = 0;
    while (alpha < beta && color < COLORS) {
        memcpy(copy_board, myboard, sizeof(copy_board));
        if (color != me && color != op) {
            prepare_fill();
            fill(copy_board, me_lin, me_col, copy_board[me_lin + 1][me_col + 1], color);
            val = -minimax(copy_board, depth - 1, -player, -beta, -alpha);
            alpha = max(alpha, val);
            score = max(score, val);
        }
        color++;
    }

    return score;
}
/* }}} */

/* }}} */

int main() { /* {{{ */
    int lin, col;
    unsigned char me, op;
    unsigned char me_lin, me_col, op_lin, op_col;
    char c;
    unsigned char color;
    int cnt, maxcnt;
    unsigned char move;

    current_player = fgetc(stdin);
    /* eat \n */
    fgetc(stdin);
    if (current_player == 'S')
        current_player = S;
    else
        current_player = J;

    /* Read the board */
    c = fgetc(stdin);
    lin = col = 0;
    while (c != EOF) {
        if (c == '\n') {
            board_cols = col;
            lin++;
            col = 0;
        } else {
            board[lin + 1][col + 1] = char_to_color(c);
            col++;
        }
        c = fgetc(stdin);
    }
    board_lines = lin;
    init_board(board);
    debug("board is %d lines %d cols wide\n", board_lines, board_cols);

    get_corner(current_player, &me_lin, &me_col);
    get_corner(-current_player, &op_lin, &op_col);
    me = board[me_lin + 1][me_col + 1];
    op = board[op_lin + 1][op_col + 1];
    debug("i'm %c/%d (%d, %d) and he is %c/%d (%d, %d)\n", color_to_char(me), current_player, me_lin, me_col, color_to_char(op), -current_player, op_lin, op_col);

    maxcnt = -INT_INFINITY;
    move = COLORS;
    /* Try all the colors */
    for (color = 0; color < COLORS; color++) {
        /* We can't move with opponent's color or with ours */
        /* Clone the board for biological experiments */
        memcpy(alt_board, board, sizeof(alt_board));
        if (color != me && color != op) {
            /* Do the move */
            prepare_fill();
            fill(alt_board, me_lin, me_col, alt_board[me_lin + 1][me_col + 1], color);
            cnt = -minimax(alt_board, DEPTH, -current_player, -INT_INFINITY, +INT_INFINITY);
            debug("[%c] minimax returned %d\n", color_to_char(color), cnt);
            if (cnt > maxcnt) {
                maxcnt = cnt;
                move = color;
            }
            debug("\n");
        }
    }
    crash(move != COLORS);
    /* Do the move */
    fill(board, me_lin, me_col, board[me_lin + 1][me_col + 1], move);
    /* Output the name of the opponent */
    if (current_player == S)
        fputc('J', stdout);
    else if (current_player == J)
        fputc('S', stdout);
    fputc('\n', stdout);
    /* Output board */
    for (lin = 0; lin < board_lines; lin++) {
        for (col = 0; col < board_cols; col++) {
            fputc(color_to_char(board[lin + 1][col + 1]), stdout);
        }
        fputc('\n', stdout);
    }
    return 0;
}
